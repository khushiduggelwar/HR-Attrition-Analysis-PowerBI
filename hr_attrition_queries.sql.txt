-- Total Employees
SELECT COUNT(*) AS Total_Employees
FROM hr_analysis;

-- Total Attrition
SELECT COUNT(*) AS Total_Attrition
FROM hr_analysis
WHERE Attrition = 'Yes';

-- Attrition by Department
SELECT Department,
       COUNT(*) AS Attrition_Count
FROM hr_analysis
WHERE Attrition = 'Yes'
GROUP BY Department
ORDER BY Attrition_Count DESC;

-- Attrition by Job Role
SELECT JobRole,
       COUNT(*) AS Attrition_Count
FROM hr_analysis
WHERE Attrition = 'Yes'
GROUP BY JobRole
ORDER BY Attrition_Count DESC;

-- Attrition by Overtime
SELECT OverTime,
       COUNT(*) AS Attrition_Count
FROM hr_analysis
WHERE Attrition = 'Yes'
GROUP BY OverTime;
